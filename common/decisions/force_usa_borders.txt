force_historical_usa_borders = {
    is_shown = {
        is_ai = no
    }
    
    possible = {
        always = yes
    }
    
    when_taken = {
        # Costa Oeste y Suroeste (Cesión Mexicana 1848)
        s:STATE_CALIFORNIA = { every_scope_state = { set_state_owner = c:USA } }
        s:STATE_TEXAS = { every_scope_state = { set_state_owner = c:USA } }
        s:STATE_NEW_MEXICO = { every_scope_state = { set_state_owner = c:USA } }
        s:STATE_ARIZONA = { every_scope_state = { set_state_owner = c:USA } }
        s:STATE_UTAH = { every_scope_state = { set_state_owner = c:USA } }
        s:STATE_NEVADA = { every_scope_state = { set_state_owner = c:USA } }
        s:STATE_COLORADO = { every_scope_state = { set_state_owner = c:USA } }
        
        # Tratado de Oregón (1846)
        s:STATE_OREGON = { every_scope_state = { set_state_owner = c:USA } }
        s:STATE_WASHINGTON = { every_scope_state = { set_state_owner = c:USA } }
        s:STATE_IDAHO = { every_scope_state = { set_state_owner = c:USA } }
        s:STATE_MONTANA = { every_scope_state = { set_state_owner = c:USA } }
        s:STATE_WYOMING = { every_scope_state = { set_state_owner = c:USA } }
        
        # Grandes Llanuras
        s:STATE_NORTH_DAKOTA = { every_scope_state = { set_state_owner = c:USA } }
        s:STATE_SOUTH_DAKOTA = { every_scope_state = { set_state_owner = c:USA } }
        s:STATE_NEBRASKA = { every_scope_state = { set_state_owner = c:USA } }
        s:STATE_KANSAS = { every_scope_state = { set_state_owner = c:USA } }
        s:STATE_OKLAHOMA = { every_scope_state = { set_state_owner = c:USA } }
        s:STATE_MINNESOTA = { every_scope_state = { set_state_owner = c:USA } }
        
        # Compra de Alaska (1867) y Hawaii (1898)
        s:STATE_ALASKA = { every_scope_state = { set_state_owner = c:USA } }
        s:STATE_HAWAIIAN_ISLANDS = { every_scope_state = { set_state_owner = c:USA } }

        # Anexión de los Estados Confederados si existen
        if = {
            limit = { exists = c:CSA }
            c:USA = {
                annex = c:CSA
            }
        }

        # Cambiar leyes y aplicar buff
        c:USA = {
            activate_law = law_type:law_presidential_republic
            activate_law = law_type:law_national_militia
            activate_law = law_type:law_interventionism
            activate_law = law_type:law_public_health_insurance
            activate_law = law_type:law_secret_police
            activate_law = law_type:law_slavery_banned
            activate_law = law_type:law_no_migration_controls

            # Modificador "Impulso Imperial"
            add_modifier = {
                name = impulse_buff_inmigracion
                months = 12
            }
        }
    }
    
    ai_chance = {
        base = 0
    }
}
