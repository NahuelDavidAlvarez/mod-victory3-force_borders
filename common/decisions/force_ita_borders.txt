    is_shown = {
        is_ai = no
    }
    
    possible = {

    }

        # Anexionar todos los estados de la región italiana a SAR primero
        s:STATE_PIEDMONT = { every_scope_state = { set_state_owner = c:SAR } }
        s:STATE_SARDINIA = { every_scope_state = { set_state_owner = c:SAR } }
        s:STATE_SAVOY = { every_scope_state = { set_state_owner = c:SAR } }
        # Niza usualmente es parte de SAVOY o PIEDMONT en el engine, pero aseguramos estados vecinos
        
        s:STATE_LOMBARDY = { every_scope_state = { set_state_owner = c:SAR } }
        s:STATE_VENETIA = { every_scope_state = { set_state_owner = c:SAR } }
        s:STATE_TUSCANY = { every_scope_state = { set_state_owner = c:SAR } }
        s:STATE_EMILIA = { every_scope_state = { set_state_owner = c:SAR } }
        s:STATE_ROMAGNA = { every_scope_state = { set_state_owner = c:SAR } }
        s:STATE_UMBRIA = { every_scope_state = { set_state_owner = c:SAR } }

        # Territorios Irredentos
        s:STATE_ISTRIA = { every_scope_state = { set_state_owner = c:SAR } }
        s:STATE_SOUTH_TYROL = { every_scope_state = { set_state_owner = c:SAR } }

        c:SAR = {
            change_tag = ITA

            activate_law = law_type:law_national_militia

            add_modifier = {
                name = impulse_buff_natalidad
                months = 12
            }
        }
    }
    
    ai_chance = {
        base = 0
    }
}
