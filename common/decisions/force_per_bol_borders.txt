force_historical_per_bol_borders = {
    is_shown = {
        is_ai = no
    }
    
    possible = {
        always = yes
    }
    
    when_taken = {
        # --- IDENTIFICAR EL PAÍS UNIFICADOR (PBC_UNIFIER) ---
        # Priorizamos Bolivia o Perú para ser el ancla de la Confederación
        if = { 
            limit = { exists = c:BOL } 
            c:BOL = { save_scope_as = pbc_unifier } 
        }
        else_if = { 
            limit = { exists = c:PEU } 
            c:PEU = { save_scope_as = pbc_unifier } 
        }
        else_if = { 
            limit = { exists = c:NPU } 
            c:NPU = { save_scope_as = pbc_unifier } 
        }
        else_if = { 
            limit = { exists = c:SPU } 
            c:SPU = { save_scope_as = pbc_unifier } 
        }
        else_if = { 
            limit = { exists = c:PBC } 
            c:PBC = { save_scope_as = pbc_unifier } 
        }
        
        # --- TRANSFERENCIA TOTAL DE TERRITORIO (SIN HUECOS) ---
        # Usamos el scope guardado para asegurar que la tierra vaya al país correcto
        if = {
            limit = { exists = scope:pbc_unifier }
            
            # Peru
            s:STATE_LIMA = { every_scope_state = { set_state_owner = scope:pbc_unifier } }
            s:STATE_ICA = { every_scope_state = { set_state_owner = scope:pbc_unifier } }
            s:STATE_AREQUIPA = { every_scope_state = { set_state_owner = scope:pbc_unifier } }
            s:STATE_CAJAMARCA = { every_scope_state = { set_state_owner = scope:pbc_unifier } }
            s:STATE_PASTAZA = { every_scope_state = { set_state_owner = scope:pbc_unifier } }
            s:STATE_TARAPACA = { every_scope_state = { set_state_owner = scope:pbc_unifier } }
            
            # Bolivia + Disputados (Acre, Antofagasta, Jujuy)
            s:STATE_LA_PAZ = { every_scope_state = { set_state_owner = scope:pbc_unifier } }
            s:STATE_POTOSI = { every_scope_state = { set_state_owner = scope:pbc_unifier } }
            s:STATE_SANTA_CRUZ = { every_scope_state = { set_state_owner = scope:pbc_unifier } }
            s:STATE_ACRE = { every_scope_state = { set_state_owner = scope:pbc_unifier } }
            s:STATE_ANTOFAGASTA = { every_scope_state = { set_state_owner = scope:pbc_unifier } }

            # --- CONSOLIDACIÓN POLÍTICA Y CULTURAL ---
            scope:pbc_unifier = {
                # Anexionar etiquetas competidoras
                if = { limit = { exists = c:BOL AND = { NOT = { this = c:BOL } } } annex = c:BOL }
                if = { limit = { exists = c:PEU AND = { NOT = { this = c:PEU } } } annex = c:PEU }
                if = { limit = { exists = c:NPU AND = { NOT = { this = c:NPU } } } annex = c:NPU }
                if = { limit = { exists = c:SPU AND = { NOT = { this = c:SPU } } } annex = c:SPU }
                if = { limit = { exists = c:PBC AND = { NOT = { this = c:PBC } } } annex = c:PBC }

                # Transformación final en la Confederación
                if = { limit = { NOT = { this ?= c:PBC } } change_tag = PBC }

                # Estabilidad y Homelands (Asegurar que la tierra sea núcleo)
                every_scope_state = {
                    add_homeland = south_andean
                    add_homeland = quechua
                    add_homeland = aimara
                }

                # Modernización y Buffs (Ajustes del Usuario)
                activate_law = law_type:law_interventionism
                activate_law = law_type:law_presidential_republic
                add_modifier = {
                    name = impulse_buff_inmigracion
                    months = 60
                }
            }
        }
    }
    
    ai_will_do = {
        base = 0
    }
}
